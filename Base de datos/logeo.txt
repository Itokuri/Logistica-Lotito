CREATE TABLE registro (
usuario VARCHAR(200),
nombre VARCHAR(200),    
apellido VARCHAR(200),        
telefono VARCHAR(200),        
contraseña VARCHAR(200),        
confirmar_contraseña VARCHAR(200),        
ID int(11),    
PRIMARY key (id)
);


CREATE TABLE logeo (
    usuario varchar(200),
    contraseña varchar(200),
    PRIMARY KEY (usuario),
    FOREIGN KEY registro REFERENCES registro(ID)
    
);
